SELECT  Distinct Voice_Actor.Voc_name
FROM VOICE, VOICE_ACTOR, CHARACTERS
WHERE VOICE_ACTOR.VOC_ID = VOICE.VOC_ID
 AND CHARACTERS.CHAR_ID=VOICE.CHAR_ID
 AND Characters.Char_perc <
 (SELECT AVG(Characters.Char_perc)
 FROM Characters,VOICE_ACTOR, Voice
 Where VOICE_ACTOR.VOC_ID = VOICE.VOC_ID
 AND CHARACTERS.CHAR_ID=VOICE.CHAR_ID
 AND VOICE_ACTOR.VOC_GENDER = 'm');
