SELECT DISTINCT characters.char_name, Count(Skins.Char_ID) AS Amount_OF_Skins 
FROM CHARACTERS, SKINS, ORIGIN,ORIGIN_PLACES
WHERE SKINS.CHAR_ID = CHARACTERS.CHAR_ID
AND ORIGIN.CHAR_ID = CHARACTERS.CHAR_ID
AND ORIGIN_PLACES.ORIGIN_ID = ORIGIN.ORIGIN_ID
AND ORIGIN_PLACES.ORIGIN_NAME= 'Frelijord'
GROUP BY CHARACTERS.CHAR_NAME;
