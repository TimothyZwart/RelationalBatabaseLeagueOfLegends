SELECT DISTINCT  Ability.Abil_name, Characters.Char_name, ABILITY_UTILITY_NAME.ABIL_UTIL_NAME
FROM ABILITY FULL join ABILITY_UTILITY on Ability.ABIL_NUM= ABILITY_UTILITY.ABIL_NUM
FULL join ABILITY_UTILITY_NAME on ABILITY_UTILITY_NAME.ABIL_UTIL_CODE= ABILITY_UTILITY.ABIL_UTIL_CODE
Full Join Characters ON Characters.CHAR_ID = Ability.CHAR_ID
AND Characters.CHAR_perc > 4
WHERE Characters.CHAR_GENDER='f'
Order by Characters.Char_name;
